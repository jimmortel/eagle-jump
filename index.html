const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); canvas.width = 320; canvas.height = 480; let eagle = { x: 140, y: 200, width: 40, height: 30, vy: 0, vx: 0, // Ajout de la vitesse horizontale jumpPower: -10, gravity: 0.4 }; let platforms = []; let score = 0; let gameRunning = true; for (let i = 0; i < 7; i++) { platforms.push({ x: Math.random() * (canvas.width - 60), y: canvas.height - (i * 80), width: 60, height: 12 }); } // Contrôles tactiles fluides let moveLeft = false; let moveRight = false; canvas.addEventListener('touchstart', (e) => { const touchX = e.touches[0].clientX; if (touchX < window.innerWidth / 2) moveLeft = true; else moveRight = true; e.preventDefault(); }); canvas.addEventListener('touchend', () => { moveLeft = false; moveRight = false; }); function update() { // Physique horizontale (inertie) if (moveLeft) eagle.vx = -5; else if (moveRight) eagle.vx = 5; else eagle.vx *= 0.9; // Ralentissement progressif eagle.x += eagle.vx; eagle.vy += eagle.gravity; eagle.y += eagle.vy; // Sortie d'écran (Wraparound) if (eagle.x + eagle.width < 0) eagle.x = canvas.width; if (eagle.x > canvas.width) eagle.x = -eagle.width; // Rebond if (eagle.vy > 0) { platforms.forEach(p => { if (eagle.x + eagle.width > p.x && eagle.x < p.x + p.width && eagle.y + eagle.height > p.y && eagle.y + eagle.height < p.y + p.height + eagle.vy) { eagle.vy = eagle.jumpPower; } }); } // Caméra if (eagle.y < canvas.height / 2) { let diff = (canvas.height / 2) - eagle.y; eagle.y = canvas.height / 2; score += Math.floor(diff / 10); platforms.forEach(p => { p.y += diff; if (p.y > canvas.height) { p.y = 0; p.x = Math.random() * (canvas.width - 60); } }); } if (eagle.y > canvas.height) { eagle.y = 200; eagle.vy = 0; eagle.vx = 0; score = 0; } draw(); requestAnimationFrame(update); } function draw() { // Fond dégradé (Aube) let gradient = ctx.createLinearGradient(0, 0, 0, canvas.height); gradient.addColorStop(0, "#87CEEB"); gradient.addColorStop(1, "#E0F6FF"); ctx.fillStyle = gradient; ctx.fillRect(0, 0, canvas.width, canvas.height); // Dessin Aigle (Style Stylisé) ctx.fillStyle = "#333"; ctx.beginPath(); ctx.moveTo(eagle.x + eagle.width/2, eagle.y); // Tête ctx.lineTo(eagle.x + eagle.width, eagle.y + eagle.height); // Aile droite ctx.lineTo(eagle.x, eagle.y + eagle.height); // Aile gauche ctx.fill(); // Dessin Ailes (Effet relief) platforms.forEach(p => { ctx.fillStyle = "#7F8C8D"; // Ombre ctx.fillRect(p.x, p.y + 4, p.width, p.height); ctx.fillStyle = "#BDC3C7"; // Face ctx.fillRect(p.x, p.y, p.width, p.height); }); // Score HUD ctx.fillStyle = "#2C3E50"; ctx.font = "bold 20px Arial"; ctx.fillText(score + " m", 20, 40); } update();
